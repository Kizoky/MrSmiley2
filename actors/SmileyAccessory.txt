//Items that the player can pickup.

ACTOR Cola : Health replaces HealthBonus
{
  +COUNTITEM
  +INVENTORY.ALWAYSPICKUP
  Inventory.Amount 1
  Inventory.MaxAmount 200
  Inventory.PickupMessage "$GOTHTHBONUS" // "Picked up a health bonus."
  States
  {
  Spawn:
    BON1 ABCDCB 6
    Loop
  }
}
