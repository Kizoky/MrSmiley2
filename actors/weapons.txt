ACTOR SmileyPistol : Pistol replaces Pistol
{
	Decal bulletchip
	Inventory.PickupMessage "Ya got the..Pistol?"
	Weapon.BobRangeX 0.4
	Weapon.BobRangeY 1.0
	Weapon.BobSpeed 1.6
	Weapon.BobStyle InverseSmooth
	States
	{
	Ready:
		PISG A 1 A_WeaponReady
		Loop
		Deselect:
		PISG A 1 A_Lower
		Loop
	Select:
		PISG A 1 A_Raise
		Loop
	Fire:
		PISG A 4
		PISG B 2 A_FirePistol
		PISG CEDB 2
		PISG A 5 A_ReFire
		Goto Ready
	Flash:
		TNT1 A 0 Bright A_Light2
		PISF A 1 Bright A_Light1
		Goto LightDone
	}
}
