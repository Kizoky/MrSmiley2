//All player actors uses different weapons so we had to do this

ACTOR ChainsawPickup : CustomInventory replaces Chainsaw
{
  Inventory.PickupMessage "$GOTCHAINSAW"
  Inventory.PickupSound "misc/w_pkup"
  States
  {
  Spawn:
    BFUG A -1
    loop
	
  Pickup:
	TNT1 A 0 A_JumpIfInventory("IsSmileyHead", 1, "PickupSmileyHead")
	TNT1 A 0 A_JumpIfInventory("IsUncleSmiley", 1, "PickupUncleSmiley")
	stop
	
  PickupSmileyHead:
	TNT1 A 0 A_GiveInventory("Toothbrush", 1)
	stop
	
  PickupUncleSmiley:
	TNT1 A 0 A_GiveInventory("Toothbrush", 1)
	stop
  }
}

}
