#include "zcommon.acs"

global int 1:monstadeaths;

script "ToastyStart" OPEN
{
	Delay(1);
	monstadeaths = 0;
}
script "ToastyChecker" OPEN
{
	Delay(75);
	if (monstadeaths > 5)
	{
		Print(s:"6 Deaths my boy! (it's working)");
		PlaySound(0,"toasty");
		monstadeaths = 0;
	}
	ACS_NamedExecute("ToastyChecker", 0);
}

script "ToastyAdder" (void)
{
	Delay(1);
	monstadeaths+1;
}
